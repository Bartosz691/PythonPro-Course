1. modyfikacja kodu produkty/serializers.py:

from rest_framework import serializers

from .models import Produkt, Note, Author, Book


class ProduktSerializer(serializers.ModelSerializer):
    class Meta:
        model = Produkt
        fields = "__all__"

#sprawdzanie czy tytul notatki  ma co najmniej 5 znaków
class NoteSerializer(serializers.ModelSerializer):

    def validate_title(self, value):
        if len(value) < 5:
            raise serializers.ValidationError(
                "Tytuł notatki musi mieć co najmniej 5 znaków."
            )

        return value

    class Meta:
        model = Note
        fields = "__all__"


class AuthorSerializer(serializers.ModelSerializer):
    class Meta:
        model = Author
        fields = "__all__"


class BookSerializer(serializers.ModelSerializer):
    author_name = serializers.CharField(
        source="author.name",
        read_only=True
    )

    class Meta:
        model = Book
        fields = [
            "id",
            "title",
            "publication_year",
            "author",
            "author_name",
        ]

def home_view(request):
    posts = Post.objects.order_by("-publication_date")

    query = request.GET.get("q")

    if query:
        posts = posts.filter(
            Q(title__icontains=query) |
            Q(content__icontains=query)
        )

    posts = posts[:5]

    return render(
        request,
        "blog/home.html",
        {
            "posts": posts,
            "query": query or "",
        }
    )
